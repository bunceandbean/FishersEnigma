function showMenu(alt){
  document.querySelector(".content-perm").style = (alt)? "display:none;" : "display:inline-block";
  let hhgreg = document.querySelectorAll(".row-display");
  for(var i = 1; i < hhgreg.length; i++){

}
  function display(group){
    let currentDisplay = [];
    let rouge = document.querySelectorAll(".showing");
    for(var i =0; i < rouge.length; i++){
      rouge[i].classList.toggle("showing");
    }
    switch (group) {
      case 'home':
        currentDisplay.push('.main-logo');
        currentDisplay.push('.main-logo');
        currentDisplay.push('.input-area');
      break;
      case 'about':
          currentDisplay.push('#abouts');
      break;
      case 'events':
          currentDisplay.push('#information');
      break;
      case 'leaderboard':
      // document.getElementById("leaderboard-frame").style.display = "inline";
      // document.getElementById("load-board").innerHTML = loadingContent;
      // document.getElementById("load-board").style.paddingTop = "400px";
      break;


    }
    let j =0;
    for(var k =0; k < currentDisplay.length; k++){
      let multi=  document.querySelectorAll(currentDisplay[k]);
      if(multi.length == 1){
      multi[0].classList.toggle("showing");
    }else{
      multi[j].classList.toggle("showing");
      j++;
    }
    }
  }
